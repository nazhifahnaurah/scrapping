  function add_update_kinerja_modal(id_modal_target, btn_add_kinerja_class){
      $(btn_add_kinerja_class).each(function(i, obj){
        $(obj).click(function(){
          var url = "https://remunerasidosen.unila.ac.id/manajemen/index.php/api_kinerja/modal_update_kinerja";

              });
            $(id_modal_target).modal('show');
        })
      });

   function add_update_rekomendasi_modal(id_modal_target, class_btn_update){
      
      $("#btn-update_rekomendasi").click(function(){
        send_form_request("#modal_form-update_rekomendasi");
      });

      $(class_btn_update).each(function(i, obj){
        $(obj).click(function(){
           $("#modal_form-update_rekomendasi input").first().val($(obj).attr('id_item'));
           $("#modal_form-update_rekomendasi select").first().val($(obj).attr('rekomendasi'));
           if(parseInt($(obj).attr('non-remun')) == 1){
            // $("#modal_form-update_rekomendasi select option").last().hide();
			$("#modal_form-update_rekomendasi option[value=" + 3 + "]").hide();
			$("#modal_form-update_rekomendasi option[value=" + 4 + "]").hide();
			$("#modal_form-update_rekomendasi option[value=" + 5 + "]").hide();
           }else{
            // $("#modal_form-update_rekomendasi select option").last().show();
			$("#modal_form-update_rekomendasi option[value=" + 3 + "]").show();
			$("#modal_form-update_rekomendasi option[value=" + 4 + "]").show();
			$("#modal_form-update_rekomendasi option[value=" + 5 + "]").show();
           }
           $(id_modal_target).modal('show');
        })
      });

    $("#btn-ubah_password").click(function(){

      $("#modal_ubah_password").modal('show');  
    })
    $("#pw_1").keyup(function(){
        onKeyup();
    });
    $("#pw_2").keyup(function(){
        onKeyup();
    });

    function onKeyup(){
        if($("#pw_1").val() != $("#pw_2").val()){
            $("#pw_1").parent().addClass('has-error');
            $("#pw_2").parent().addClass('has-error');
            $("#status_ubah_password").addClass('bg-red');
            $("#status_ubah_password").removeClass('bg-green');
            $("#status_ubah_password").text("Password Baru Tidak Sesuai");
            $("#btn-ubah_password").attr("disabled","true");
        }else{
            $("#pw_1").parent().removeClass('has-error');
            $("#pw_2").parent().removeClass('has-error');            
            $("#status_ubah_password").addClass('bg-green');
            $("#status_ubah_password").removeClass('bg-red');
            $("#status_ubah_password").text("Password Baru Sesuai");
            $("#btn-ubah_password").removeAttr("disabled");
        }

    }

    var old_ask_unread = 0;
    function sync_ask(sync_pesan_url){
        var interval = 5000;
        $.ajax({
          type: "POST",
          url: sync_pesan_url,
          dataType: "json",
          success: function(data) {
                // log(data.unread);
                if(data.unread > 0){
                    $("#ask-status").show(800);                    
                    $("#ask-status").text(data.unread+" pesan baru");
                    if(old_ask_unread  < data.unread){
                        old_ask_unread  = data.unread;
                        notify('top', 'right', '', 'inverse', 'animated fadeIn', 'animated fadeOut', data.unread, ' Pesan Baru');                        
                    }
                }else{
                    $("#ask-status").hide(800);                    
                }
                setTimeout(function(){
                    // sync_ask(sync_pesan_url);
                }, interval);
            },
            error: function(data) {
                log(data.responseText);
                setTimeout(function(){
                    // sync_ask(sync_pesan_url);
                }, interval);
                
            }
        });

    }
